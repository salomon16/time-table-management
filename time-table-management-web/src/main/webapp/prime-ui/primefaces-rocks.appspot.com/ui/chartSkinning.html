Charts Skinning

In this example CPU and Memory usages of four different servers are monitored live and displayed using advanced skinning options.

Source

  • chartSkinning.xhtml
  • ChartBean.java
  • ChartStyle
<h:form>
    <p:columnChart value="#{chartBean.servers}" var="server" 
            xfield="#{server.name}" style="chartStyle" live="true">
        <p:chartSeries label="CPU" value="27" style="cpuStyle"/>
        <p:chartSeries label="Memory" value="81" style="memoryStyle"/>
    </p:columnChart>
</h:form>
                    
public class ChartBean implements Serializable {

    private List<ServerLoad> servers;

	public ChartBean() {
        servers = new ArrayList<ServerLoad>(5);
        for (int i = 0; i < 4; i++) {
            servers.add(i, new ServerLoad("Server " + i, (int)(Math.random() * 100), (int)(Math.random() * 100)));
        }
	}

    public List<ServerLoad> getServers() {
        for (int i = 0; i < 4; i++) {
            servers.get(i).update( (int)(Math.random() * 100), (int)(Math.random() * 100));
        }

        return servers;
    }
}
