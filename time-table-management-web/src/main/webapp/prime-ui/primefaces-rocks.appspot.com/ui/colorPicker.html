Color Picker

ColorPicker is a highly customizable component that also gives a live feedback when selecting a color. By default, value is passed as a java.awt.Color, using custom converters formats like RGB can also be supported.

<h:form>
			
	<p:panel header="Colors">
		<h:panelGrid columns="3">
			<h:outputText value="AWT Color: " />
			<p:colorPicker value="#{colorBean.color}" />
			<h:outputText id="awtDisplay" 
					value="#{colorBean.color}" escape="false"/>
			
			<h:outputText value="RGB Color: " />
			<p:colorPicker value="#{colorBean.rgbColor}">
				<f:converter converterId="colorPickerConverter"/>
			</p:colorPicker>
			<h:outputText id="rgbDisplay" 
					value="\#{colorBean.rgbColor}" 
					style="background-color:rgb(#{colorBean.rgbColor})"/>
		</h:panelGrid>

        <p:commandButton value="Submit" update="awtDisplay rgbDisplay" />
	</p:panel>
	
</h:form>

package org.primefaces.examples.view;

import javax.faces.component.UIComponent;
import javax.faces.context.FacesContext;
import javax.faces.convert.Converter;

public class ColorPickerConverter implements Converter {

	public Object getAsObject(FacesContext facesContext, UIComponent component, String submittedValue) {
		return submittedValue; //just return the rgb value as string
	}

	public String getAsString(FacesContext facesContext, UIComponent component, Object value) {
		if (value == null)
			return null;
		else{
			return value.toString();
		}
	}
}
