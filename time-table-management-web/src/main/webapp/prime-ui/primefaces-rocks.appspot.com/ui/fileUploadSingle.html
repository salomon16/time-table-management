FileUpload - Single

Select a single file from file browser and click upload button. If your browser supports, you can drag and drop the file from your operation system and progress bar will be displayed.

<h:form enctype="multipart/form-data">

    <p:growl id="messages" showDetail="true"/>

    <p:fileUpload fileUploadListener="#{fileUploadController.handleFileUpload}"
            mode="advanced" 
            update="messages"
            sizeLimit="1000000"
            allowTypes="png,gif,jpg" />

</h:form>
                    
package org.primefaces.examples.view;

import javax.faces.application.FacesMessage;
import javax.faces.context.FacesContext;

import org.primefaces.event.FileUploadEvent;
import org.primefaces.model.UploadedFile;

public class FileUploadController {

    public void handleFileUpload(FileUploadEvent event) {
		FacesMessage msg = new FacesMessage("Succesful", event.getFile().getFileName() + " is uploaded.");
		FacesContext.getCurrentInstance().addMessage(null, msg);
	}
}
