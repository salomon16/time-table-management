Galleria

Galleria is an image gallery component featuring various transition effects and detail display.

Effect:

Source

  • galleria.xhtml
  • GalleriaBean.java
<p:galleria effect="#{galleriaBean.effect}" effectSpeed="1000">
    <ui:repeat value="#{galleriaBean.images}" var="image">
        <p:graphicImage value="/images/galleria/#{image}"
            title="#{image}" alt="Description goes here"/>
    </ui:repeat>
</p:galleria>
                    
package org.primefaces.examples.view;

import java.util.ArrayList;
import java.util.List;
import javax.annotation.PostConstruct;

public class GalleriaBean {

    private List<String> images;

    private String effect = "fade";

    @PostConstruct
    public void init() {
        images = new ArrayList<String>();

        for(int i=1;i<=12;i++) {
            images.add("galleria" + i + ".jpg");
        }
    }

    public List<String> getImages() {
        return images;
    }

    public String getEffect() {
        return effect;
    }

    public void setEffect(String effect) {
        this.effect = effect;
    }
}
