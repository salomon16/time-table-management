Resizable - AJAX

When resizing is completed, a server side resizeListener can be invoked and other components on page can be updated partially with ajax.

Source

  • resizeAjax.xhtml
  • ResizeBean.java
<h:form>
    <p:growl id="growl" showDetail="true" />

    <p:graphicImage id="campnou" value="/images/barca/camp_nou.jpg" />

    <p:resizable for="campnou" animate="true" ghost="true"
                resizeListener="#{resizeBean.onResize}"
                onResizeUpdate="growl" />
</h:form>
                    
package org.primefaces.examples.view;

import javax.faces.application.FacesMessage;
import javax.faces.context.FacesContext;
import org.primefaces.event.ResizeEvent;

public class ResizeBean {

    public void onResize(ResizeEvent event) {
        FacesMessage msg = new FacesMessage(FacesMessage.SEVERITY_INFO,
                        "Image resized", "Width:" + event.getWidth() + ",Height:" + event.getHeight());

        FacesContext.getCurrentInstance().addMessage(null, msg);
    }
}
