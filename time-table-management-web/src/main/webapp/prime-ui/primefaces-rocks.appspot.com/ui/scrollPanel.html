  • scrollPanel.xhtml
<h3>Basic ScrollPanel</h3>
<p:scrollPanel style="width:250px;height:200px">
    <p:dataGrid var="car" value="#{tableBean.carsSmall}" columns="3">
        <p:panel header="#{car.model}" style="text-align:center">
            <h:panelGrid columns="1" style="width:100%">
                <h:outputText value="#{car.model}" />
                <h:outputText value="#{car.manufacturer}" />
                <h:outputText value="#{car.year}" />
                <h:outputText value="#{car.color}" />
            </h:panelGrid>
        </p:panel>
    </p:dataGrid>
</p:scrollPanel>

<h3>Button Scroll</h3>
<p:scrollPanel style="width:250px;height:200px" scrollMode="buttons">
    //same datagrid as in basic example
</p:scrollPanel>

<h3>Edge Scroll</h3>
<p:scrollPanel style="width:250px;height:200px" scrollMode="edge">
    //same datagrid as in basic example
</p:scrollPanel>

<h3>Custom Animation</h3>
<p:scrollPanel style="width:250px;height:200px" easing="easeOutElastic">
    //same datagrid as in basic example
</p:scrollPanel>

<h3>Client Side API</h3>
<p:commandButton type="button" value="Scroll" onclick="scroller.scrollTo(100, 100)"
        image="ui-icon-arrowthick-1-se"/>

<p:scrollPanel style="width:250px;height:200px" widgetVar="scroller">
    //same datagrid as in basic example
</p:scrollPanel>
