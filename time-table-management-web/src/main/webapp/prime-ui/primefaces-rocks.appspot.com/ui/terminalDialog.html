Terminal - Dialog

This example demonstrates dialog integration.

Source

  • terminal.xhtml
  • TerminalController.java
<h:form>
    <p:graphicImage value="/images/terminal.png"
            onclick="dlg.show();term.focus();"
            style="cursor:pointer"/>

    <p:dialog widgetVar="dlg" width="600" height="400" header="Terminal" 
        resizable="false" modal="true">

        <p:terminal width="100%" height="100%"
                    commandHandler="#{terminalController.handleCommand}"
                    welcomeMessage="Welcome to PrimeFaces Terminal, how are you today?"/>

    </p:dialog>
</h:form>
                    
package org.primefaces.examples.view;

import java.util.Date;

public class TerminalController {

	public String handleCommand(String command, String[] params) {
		if(command.equals("greet"))
			return "Hello " + params[0];
		else if(command.equals("date"))
			return new Date().toString();
		else
			return command + " not found";
	}
}
