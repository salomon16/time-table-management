Tooltip - Configuration

Tooltip has various customization options such as positioning, delay, effects and events.

Delayed: Longer
Focus:
Fade: Fade
Slide: Slide
Grow: Grow
Position: TopRight
Content: PrimeFaces

Source

  • tooltipOptions.xhtml
<h:panelGrid columns="3" cellpadding="5">

    <h:outputText value="Delayed: " />
    <h:outputLink id="longerDelay" value="#">
        <h:outputText value="Longer" />
    </h:outputLink>
    <p:tooltip for="longerDelay" value="Show and hide delays are extended for this tip" showDelay="2000" hideDelay="2000"/>

    <h:outputText value="Focus: " />
    <p:inputText id="focus" />
    <p:tooltip for="focus" value="This tooltip is displayed when input gets the focus"
               showEvent="focus" hideEvent="blur"/>

    <h:outputText value="Fade: " />
    <h:outputLink id="fade" value="#">
        <h:outputText value="Fade" />
    </h:outputLink>
    <p:tooltip for="fade" value="Fade effect is used by default"
               showEffect="fade" hideEffect="fade" showEffectLength="1000" hideEffectLength="1000"/>

    <h:outputText value="Slide: " />
    <h:outputLink id="slide" value="#">
        <h:outputText value="Slide" />
    </h:outputLink>
    <p:tooltip for="slide" value="This tooltip uses slide effect for the animation"
                        showEffect="slide" hideEffect="slide"
                        showEffectLength="500" hideEffectLength="500"/>

    <h:outputText value="Grow: " />
    <h:outputLink id="grow" value="#">
        <h:outputText value="Grow" />
    </h:outputLink>
    <p:tooltip for="grow" value="This tooltip uses grow effect for the animation"
            showEffect="grow" hideEffect="grow" showEffectLength="500" hideEffectLength="500" />

    <h:outputText value="Position:  " />
    <h:outputLink id="topleft" value="#">
        <h:outputText value="TopRight" />
    </h:outputLink>
    <p:tooltip for="topleft" targetPosition="topRight" position="bottomLeft" value="This tooltip is positioned at top right" />

    <h:outputText value="Content:  " />
    <h:outputLink id="lnk" value="#">
        <h:outputText value="PrimeFaces" />
    </h:outputLink>
    <p:tooltip for="lnk">
        <p:graphicImage value="/images/prime_logo.png" />
    </p:tooltip>

</h:panelGrid>
